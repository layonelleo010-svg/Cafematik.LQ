Control de Inventario

Utilice el buscador para consultar o agregar activos.

`; } // ============================================ // 2. LÓGICA DE NEGOCIO (funciones del servidor) // ============================================ // Buscar código y mostrar info / formulario para agregar si no existe function procesarBusqueda(codigo) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const master = ss.getSheetByName(SHEET_MAESTRO); const data = master.getDataRange().getValues(); // Buscar en maestro let encontrado = null; for (let i = 1; i < data.length; i++) { if (data[i][0].toString() === codigo) { encontrado = data[i]; break; } } if (encontrado) { return construirVistaActivo(encontrado); } else { // No existe en maestro → ofrecer agregar (se guardará en Nuevos) return `

Activo no encontrado

¿Desea agregar el código ${codigo} a la lista de nuevos?

`; } } function construirVistaActivo(fila) { const [codigo, , , , nombre, desc, , , , , , estado, codUbic, nomUbic, , nomResp, , , , , , , , nuevaUbic, nuevoResp] = fila; const ubicOrig = nomUbic ? nomUbic : (codUbic ? codUbic : 'No definida'); return `

Código: ${codigo}

Nombre: ${nombre}

Descripción: ${desc}

Ubicación original: ${ubicOrig}

Responsable original: ${nomResp}

Estado original: ${estado}

Nueva ubicación: ${nuevaUbic || '—'}

Nuevo responsable: ${nuevoResp || '—'}

Para registrar un movimiento, utilice la opción correspondiente.

`; } // Agregar nuevo activo a la hoja "Nuevos" function agregarNuevoActivo(codigo) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const nuevos = ss.getSheetByName(SHEET_NUEVOS); if (!nuevos) { throw new Error('No existe la hoja "Nuevos". Cree la segunda pestaña con ese nombre.'); } // Obtener último código en la columna A de Nuevos para autoincremento const data = nuevos.getDataRange().getValues(); let ultimoCod = ""; if (data.length > 1) { ultimoCod = data[data.length-1][0].toString(); } // Si no hay datos, el nuevo código será el mismo que se busca (ya es un código válido) // Opcionalmente se podría generar automáticamente, pero según tu indicación se debe usar el código ingresado. // Verificar que el código no exista ya en Nuevos const existe = data.some(fila => fila[0].toString() === codigo); if (existe) { return `El código ${codigo} ya fue agregado anteriormente.`; } // Insertar nueva fila al final nuevos.appendRow([codigo, '', '', '', '', '', '', '', '', '', '', '', '', '', '', '', '', '', '', '', '', '', '', '', '']); // Ajusta según las columnas necesarias // Registrar en historial registrarHistorial(codigo, "AGREGADO", "Nuevo activo agregado a la lista de Nuevos"); return `✅ Activo ${codigo} agregado correctamente a la pestaña "Nuevos".`; } // ============================================ // 3. HISTORIAL DESCENDENTE Y NO ACUMULATIVO // ============================================ function registrarHistorial(codigo, accion, detalle) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const hist = ss.getSheetByName(SHEET_HISTORIAL); if (!hist) { ss.insertSheet(SHEET_HISTORIAL); ss.getSheetByName(SHEET_HISTORIAL).appendRow(["Fecha", "Código", "Acción", "Detalle"]); } // Insertar al principio para que siempre veamos descendente (más reciente arriba) hist.insertRowBefore(2); hist.getRange(2, 1, 1, 4).setValues([[new Date(), codigo, accion, detalle]]); } // Exportar historial: copiar la hoja Historial en un nuevo libro o limpiar y volcar ordenado function exportarHistorial() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const hist = ss.getSheetByName(SHEET_HISTORIAL); if (!hist || hist.getLastRow() <= 1) { return 'No hay historial para exportar.'; } // Crear nueva hoja temporal ordenada descendente (ya está insertada así, pero aseguramos) const datos = hist.getDataRange().getValues(); // Ordenar descendentemente por fecha (columna 0) si fuera necesario, pero como insertamos arriba, ya lo está. // Crear un nuevo libro con los datos del historial const nuevoLibro = SpreadsheetApp.create("Historial_exportado_" + Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "yyyyMMdd_HHmmss")); const hojaExp = nuevoLibro.getSheets()[0]; hojaExp.getRange(1, 1, datos.length, datos[0].length).setValues(datos); return 'Historial exportado exitosamente.'; } // ============================================ // 4. FUNCIONES AUXILIARES PARA MANTENER LA HOJA PROGRESO (opcional) // ============================================ // (Si necesitas regenerar la hoja Progreso con ubicación original, puedes llamar a generarProgreso desde un menú) function onOpen() { const ui = SpreadsheetApp.getUi(); ui.createMenu('Control Inventario') .addItem('Generar hoja Progreso', 'generarProgreso') .addItem('Exportar historial', 'exportarHistorial') .addToUi(); } function generarProgreso() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const maestro = ss.getSheetByName(SHEET_MAESTRO); if (!maestro) { throw new Error('No existe la hoja "DATOS MAESTROS"'); } const datos = maestro.getDataRange().getValues(); const progreso = []; progreso.push(["CODIGOAF","NOMBREAF","DESCAF","UBICACION ORIGINAL","RESPONSABLE ORIGINAL","ESTADO ORIGINAL","NUEVA UBICACION","NUEVO RESPONSABLE","NUEVO ESTADO","OBSERVACIONES","ESTADO INVENTARIO","FECHA INVENTARIO"]); for (let i = 1; i < datos.length; i++) { const fila = datos[i]; const ubicOrig = fila[13]; // NOM UBICACIÓN (columna N, índice 13) const respOrig = fila[15]; // NOM RESP ANT (columna P) const estadoOrig = fila[11] || "Revalúo"; progreso.push([ fila[0], fila[4], fila[5], ubicOrig, respOrig, estadoOrig, fila[22] || "", fila[23] || "", "", fila[24] || "", "pendiente", "" ]); } let hojaProgreso = ss.getSheetByName("Progreso"); if (hojaProgreso) { ss.deleteSheet(hojaProgreso); } hojaProgreso = ss.insertSheet("Progreso"); hojaProgreso.getRange(1,1,progreso.length, progreso[0].length).setValues(progreso); SpreadsheetApp.getUi().alert('Hoja Progreso generada correctamente.'); }
